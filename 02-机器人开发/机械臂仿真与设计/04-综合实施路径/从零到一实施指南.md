### 2.1 线性代数复习

#### 齐次变换矩阵

```
齐次变换矩阵是描述刚体位姿的标准方法

T = ┌ R   p ┐
    │       │
    │ 0   1 │
    └       ┘

其中：
- R（3×3）：旋转矩阵
- p（3×1）：位置向量

代码实现：
```

```python
import numpy as np

def homogeneous_transform(R, p):
    """
    构造齐次变换矩阵

    参数：
        R: 旋转矩阵 (3,3)
        p: 位置向量 (3,)

    返回：
        T: 齐次变换矩阵 (4,4)
    """
    T = np.eye(4)
    T[:3, :3] = R
    T[:3, 3] = p
    return T

def rotation_z(theta):
    """绕Z轴旋转"""
    ct, st = np.cos(theta), np.sin(theta)
    return np.array([
        [ct, -st, 0],
        [st,  ct, 0],
        [0,   0,  1]
    ])

def rotation_x(theta):
    """绕X轴旋转"""
    ct, st = np.cos(theta), np.sin(theta)
    return np.array([
        [1,  0,   0],
        [0, ct, -st],
        [0, st,  ct]
    ])

def rpy_to_rotation(roll, pitch, yaw):
    """
    RPY角转旋转矩阵（ZYX顺序）

    R = Rz(yaw) × Ry(pitch) × Rx(roll)
    """
    Rz = rotation_z(yaw)
    Ry = rotation_y(pitch)
    Rx = rotation_x(roll)
    return Rz @ Ry @ Rx
```

### 2.2 DH参数与正运动学

#### DH法则详解

```
标准DH法则（Standard DH）：

坐标系建立步骤：

1. 确定关节轴线 Z0, Z1, ..., Zn

2. 建立基坐标系
   - Z0：沿第一个关节轴线
   - X0、Y0：根据需要选择

3. 对于第i个坐标系（i=1到n）：
   - Zi：沿第i+1个关节轴线
   - Xi：沿Zi-1和Zi的公垂线
   - Yi：满足右手定则

4. DH参数：
   - θi：绕Zi-1的旋转角（关节变量）
   - di：沿Zi-1的距离
   - ai：沿Xi的距离
   - αi：绕Xi的旋转角
```

#### 单关节变换矩阵推导

```
单关节变换矩阵：

Ti = Rot(Zi-1, θi) × Trans(0,0,di) × Trans(ai,0,0) × Rot(Xi, αi)

展开：

Ti = ┌ cosθ  -sinθ·cosα   sinθ·sinα   a·cosθ ┐
     │ sinθ   cosθ·cosα  -cosθ·sinα   a·sinθ │
     │  0       sinα        cosα         d    │
     └  0         0           0          1    ┘
```

### 2.3 逆运动学

#### 雅可比矩阵

```
雅可比矩阵描述关节速度到末端速度的映射：

v = J × q̇

对于旋转关节i，雅可比的第i列：

J_i = [z_{i-1} × (p_e - p_{i-1})]  (线速度)
      [        z_{i-1}         ]  (角速度)
```

```python
def singularity_analysis():
    """
    奇异点分析实验

    目标：识别和可视化奇异点
    """
    robot = create_7dof_robot()

    # 在工作空间中采样，计算可操作性
    N = 10000
    manipulability = np.zeros(N)
    positions = np.zeros((N, 3))

    for i in range(N):
        q = np.random.uniform(robot.qlim[:, 0], robot.qlim[:, 1])

        # 计算雅可比
        J = robot.jacob0(q)

        # 可操作性
        manipulability[i] = np.sqrt(np.abs(np.linalg.det(J @ J.T)))

        # 位置

        positions[i] = T.t

    fig = plt.figure(figsize=(10, 8))

    ax.set_xlabel('X (m)')
    ax.set_ylabel('Y (m)')
    ax.set_zlabel('Z (m)')
```

```python
    def forward_dynamics(self, q, qd, tau):
        """正向动力学：τ → q̈"""
        M = self.mass_matrix(q)
        C = self.coriolis_matrix(q, qd)
        G = self.gravity_vector(q)
        return np.linalg.solve(M, tau - C @ qd - G)

    def inverse_dynamics(self, q, qd, qdd):
        """逆向动力学：q̈ → τ"""
        M = self.mass_matrix(q)
        C = self.coriolis_matrix(q, qd)
        G = self.gravity_vector(q)
        return M @ qdd + C @ qd + G
```

```python
def dynamics_validation():
    """
    动力学模型验证

    方法：能量守恒验证
    """
    dynamics = create_dynamics_model()

    # 初始状态
    q0 = np.array([0, 0.5, 0, -0.5, 0, 0.3, 0])
    qd0 = np.zeros(7)

    # 无外力仿真
    dt = 0.001
    T = 2.0
    n_steps = int(T / dt)

    q = q0.copy()
    qd = qd0.copy()

    energies = []

    for step in range(n_steps):
        # 计算能量
        M = dynamics.mass_matrix(q)
        K = 0.5 * qd @ M @ qd  # 动能
        P = compute_potential_energy(dynamics, q)  # 势能
        E = K + P
        energies.append(E)

        # 正向动力学（零力矩）
        qdd = dynamics.forward_dynamics(q, qd, np.zeros(7))

        # 积分
        qd = qd + qdd * dt
        q = q + qd * dt

    # 绘图
    plt.figure(figsize=(10, 6))
    plt.plot(np.linspace(0, T, n_steps), energies)
    plt.xlabel('时间 (s)')
    plt.ylabel('总能量 (J)')
    plt.title('能量守恒验证')
    plt.grid(True)
    plt.savefig('energy_conservation.png', dpi=150)
    plt.show()

    # 检查能量变化
    energy_change = (max(energies) - min(energies)) / energies[0]
    print(f"能量变化: {energy_change*100:.2f}%")
    print("能量守恒验证" + ("通过" if energy_change < 0.01 else "失败"))
```

```python
        q = q + qd * dt

        q_actual[step] = q
```

```python
def impedance_control_experiment():
    """
    阻抗控制实验

    目标：验证阻抗控制的力控性能
    """
    dynamics = create_dynamics_model()
    kin = create_kinematics()

    controller = ImpedanceController(
        dynamics, kin.compute_jacobian,
        kd=np.diag([500, 500, 500, 50, 50, 50]),
        dd=np.diag([100, 100, 100, 10, 10, 10])
    )

    # 仿真场景：末端接触平面
    x_desired = np.array([0.4, 0.2, 0.5, 0, 0, 0])
    x_surface = 0.3  # 平面高度

    q = np.zeros(7)
    qd = np.zeros(7)

    for step in range(1000):
        # 当前末端位姿
        T_ee, _ = kin.forward_kinematics(q)
        x_actual = np.concatenate([
            T_ee[:3, 3],
            kin._rotation_to_rpy(T_ee[:3, :3])
        ])

        # 计算接触力
        f_ext = np.zeros(6)
        if x_actual[2] < x_surface:
            f_ext[2] = 1000 * (x_surface - x_actual[2])  # 弹簧接触

        # 阻抗控制
        tau = controller.compute(
            q, qd,
            x_desired, np.zeros(6), np.zeros(6),
            x_actual, np.zeros(6),
            f_ext
        )

        # 动力学仿真
        qdd = dynamics.forward_dynamics(q, qd, tau)
        qd = qd + qdd * 0.001
        q = q + qd * 0.001
```
