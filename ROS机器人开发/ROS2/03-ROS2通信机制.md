# ROS2通信机制

本章详细介绍ROS2的通信方式。与ROS1相比，ROS2去掉了ROS Master，基于DDS实现去中心化通信，并引入了QoS策略。建议先完成《01-ROS2概述与环境搭建》。

---

# 一、通信架构总览

ROS2的通信不再依赖ROS Master，节点之间通过DDS（Data Distribution Service）自动发现和通信：

### ROS1 vs ROS2通信架构对比

```
      ROS1 架构                          ROS2 架构
┌──────────────────┐            ┌──────────────────────┐
│   ROS Master     │            │                      │
│  ┌────────────┐  │            │   Node A             │
│  │ 注册中心   │  │            │  (Publisher)         │
│  └─────┬──────┘  │            │        │             │
├────────┼─────────┤            │        │ DDS自动发现  │
│        │         │            │        ↓             │
│  Node A    Node B │            │   Topic: /sensor    │
│  (Pub)    (Sub)  │            │        │             │
│    │        │    │            │        ↓             │
│    └────────┘    │            │   Node B             │
│   通过Master发现  │            │  (Subscriber)       │
└──────────────────┘            └──────────────────────┘
  单点故障风险                    去中心化，更可靠
```

### DDS通信流程
```
┌──────────────┐                    ┌──────────────┐
│   Node A     │                    │   Node B     │
│  (Publisher) │    DDS Discovery   │ (Subscriber) │
│              │ ←───────────────→  │              │
└──────┬───────┘                    └──────┬───────┘
       │          Topic: /sensor           │
       └──────────────────────────────────→┘
                   (Direct P2P)
```

**核心变化：**
- 去掉了ROS Master，节点通过DDS自动发现
- 引入**域ID（Domain ID）**：不同域的节点互不干扰，适合多机器人
- 引入**QoS策略**：精细控制通信的可靠性、延迟等行为

# 二、节点（Node）

## 节点是什么？

节点是ROS2中最小的可执行单元，每个节点负责一个具体功能（如传感器驱动、运动规划等）。节点之间通过话题、服务、动作进行通信。

## Python节点示例

```python
#!/usr/bin/env python3
import rclpy
from rclpy.node import Node

class MyNode(Node):
    def __init__(self):
        super().__init__('my_node')           # 节点名称
        self.get_logger().info('节点已启动！')  # 打印日志
        self.timer = self.create_timer(1.0, self.timer_callback)  # 1秒定时器

    def timer_callback(self):
        self.get_logger().info('Hello ROS2!')

def main(args=None):
    rclpy.init(args=args)
    node = MyNode()
    rclpy.spin(node)      # 保持节点运行
    node.destroy_node()
    rclpy.shutdown()

if __name__ == '__main__':
    main()
```

## C++节点示例

```cpp
#include <rclcpp/rclcpp.hpp>
#include <chrono>

using namespace std::chrono_literals;

class MyNode : public rclcpp::Node {
public:
    MyNode() : Node("my_node") {
        RCLCPP_INFO(get_logger(), "节点已启动！");
        timer_ = create_wall_timer(1s, [this]() {
            RCLCPP_INFO(get_logger(), "Hello ROS2!");
        });
    }
private:
    rclcpp::TimerBase::SharedPtr timer_;
};

int main(int argc, char** argv) {
    rclcpp::init(argc, argv);
    rclcpp::spin(std::make_shared<MyNode>());
    rclcpp::shutdown();
}
```

# 三、话题通信（Topic）

## 发布者（Publisher）

**Python：**
```python
#!/usr/bin/env python3
import rclpy
from rclpy.node import Node
from std_msgs.msg import String

class Talker(Node):
    def __init__(self):
        super().__init__('talker')
        self.pub = self.create_publisher(String, 'chatter', 10)  # 队列大小10
        self.timer = self.create_timer(0.5, self.timer_callback)  # 2Hz
        self.count = 0

    def timer_callback(self):
        msg = String()
        msg.data = f'Hello ROS2! Count: {self.count}'
        self.pub.publish(msg)
        self.get_logger().info(f'Published: {msg.data}')
        self.count += 1

def main(args=None):
    rclpy.init(args=args)
    rclpy.spin(Talker())
    rclpy.shutdown()
```

## 订阅者（Subscriber）

**Python：**
```python
#!/usr/bin/env python3
import rclpy
from rclpy.node import Node
from std_msgs.msg import String

class Listener(Node):
    def __init__(self):
        super().__init__('listener')
        self.sub = self.create_subscription(
            String, 'chatter', self.callback, 10  # QoS队列大小10
        )

    def callback(self, msg):
        self.get_logger().info(f'Received: {msg.data}')

def main(args=None):
    rclpy.init(args=args)
    rclpy.spin(Listener())
    rclpy.shutdown()
```

## QoS配置示例

```python
from rclpy.qos import QoSProfile, ReliabilityPolicy, DurabilityPolicy, HistoryPolicy

# 传感器数据：低延迟，允许丢帧
sensor_qos = QoSProfile(
    history=HistoryPolicy.KEEP_LAST,
    depth=5,
    reliability=ReliabilityPolicy.BEST_EFFORT,
    durability=DurabilityPolicy.VOLATILE
)

# 重要控制指令：可靠传输
control_qos = QoSProfile(
    history=HistoryPolicy.KEEP_LAST,
    depth=10,
    reliability=ReliabilityPolicy.RELIABLE,
    durability=DurabilityPolicy.TRANSIENT_LOCAL
)

# 使用
self.pub = self.create_publisher(Twist, '/cmd_vel', control_qos)
```

# 四、服务通信（Service）

## 服务端（Server）

**Python：**
```python
#!/usr/bin/env python3
import rclpy
from rclpy.node import Node
from example_interfaces.srv import AddTwoInts

class AddServer(Node):
    def __init__(self):
        super().__init__('add_server')
        self.srv = self.create_service(
            AddTwoInts, 'add_two_ints', self.callback
        )

    def callback(self, request, response):
        response.sum = request.a + request.b
        self.get_logger().info(f'{request.a} + {request.b} = {response.sum}')
        return response

def main(args=None):
    rclpy.init(args=args)
    rclpy.spin(AddServer())
    rclpy.shutdown()
```

## 客户端（Client）

**Python（同步调用）：**
```python
#!/usr/bin/env python3
import rclpy
from rclpy.node import Node
from example_interfaces.srv import AddTwoInts

class AddClient(Node):
    def __init__(self):
        super().__init__('add_client')
        self.client = self.create_client(AddTwoInts, 'add_two_ints')
        while not self.client.wait_for_service(timeout_sec=1.0):
            self.get_logger().info('等待服务...')

    def send_request(self, a, b):
        request = AddTwoInts.Request()
        request.a = a
        request.b = b
        future = self.client.call_async(request)
        return future

def main(args=None):
    rclpy.init(args=args)
    node = AddClient()
    future = node.send_request(3, 5)
    rclpy.spin_until_future_complete(node, future)
    node.get_logger().info(f'Result: {future.result().sum}')
    rclpy.shutdown()
```

## 服务操作命令

```bash
# 查看所有服务
ros2 service list

# 查看服务类型
ros2 service type /add_two_ints

# 调用服务
ros2 service call /add_two_ints example_interfaces/srv/AddTwoInts "{a: 3, b: 5}"
```

# 五、动作通信（Action）

## 动作服务端

**Python：**
```python
#!/usr/bin/env python3
import rclpy
from rclpy.node import Node
from rclpy.action import ActionServer
from example_interfaces.action import Fibonacci

class FibonacciServer(Node):
    def __init__(self):
        super().__init__('fibonacci_server')
        self._action_server = ActionServer(
            self, Fibonacci, 'fibonacci', self.execute_callback
        )

    async def execute_callback(self, goal_handle):
        self.get_logger().info(f'收到目标: order={goal_handle.request.order}')
        
        feedback_msg = Fibonacci.Feedback()
        feedback_msg.sequence = [0, 1]
        
        for i in range(1, goal_handle.request.order):
            feedback_msg.sequence.append(
                feedback_msg.sequence[i] + feedback_msg.sequence[i-1]
            )
            goal_handle.publish_feedback(feedback_msg)
            self.get_logger().info(f'反馈: {feedback_msg.sequence}')
        
        goal_handle.succeed()
        result = Fibonacci.Result()
        result.sequence = feedback_msg.sequence
        return result

def main(args=None):
    rclpy.init(args=args)
    rclpy.spin(FibonacciServer())
    rclpy.shutdown()
```

## 动作客户端

**Python：**
```python
#!/usr/bin/env python3
import rclpy
from rclpy.node import Node
from rclpy.action import ActionClient
from example_interfaces.action import Fibonacci

class FibonacciClient(Node):
    def __init__(self):
        super().__init__('fibonacci_client')
        self._client = ActionClient(self, Fibonacci, 'fibonacci')

    def send_goal(self, order):
        goal_msg = Fibonacci.Goal()
        goal_msg.order = order
        self._client.wait_for_server()
        future = self._client.send_goal_async(goal_msg, feedback_callback=self.feedback_cb)
        return future

    def feedback_cb(self, feedback):
        self.get_logger().info(f'反馈: {feedback.feedback.sequence}')

def main(args=None):
    rclpy.init(args=args)
    node = FibonacciClient()
    future = node.send_goal(10)
    rclpy.spin_until_future_complete(node, future)
    rclpy.shutdown()
```

## 动作操作命令

```bash

ros2 action list

# 查看动作信息
ros2 action info /fibonacci

# 发送动作目标
ros2 action send_goal /fibonacci example_interfaces/action/Fibonacci "{order: 10}"
```

# 六、参数服务器（Parameter）

ROS2中每个节点都有自己的参数，不再有全局参数服务器。

## 在节点中使用参数

```python
class MyNode(Node):
    def __init__(self):
        super().__init__('my_node')
        # 声明参数（名称、默认值、描述）
        self.declare_parameter('robot_name', 'my_robot')
        self.declare_parameter('max_speed', 1.0)
        
        # 获取参数值
        name = self.get_parameter('robot_name').value
        speed = self.get_parameter('max_speed').value
        self.get_logger().info(f'Robot: {name}, Max speed: {speed}')
```

## 参数操作命令

```bash
# 查看所有参数
ros2 param list

# 获取参数值
ros2 param get /my_node robot_name

# 设置参数值
ros2 param set /my_node max_speed 2.0

# 导出参数到YAML文件
ros2 param dump /my_node

# 从YAML文件加载参数
ros2 run my_pkg my_node --ros-args --params-file params.yaml
```

# 七、通信方式选择指南

| 场景 | 推荐方式 | 原因 |
|------|----------|------|
| 传感器持续数据流 | 话题 | 异步、高效、支持多订阅者 |
| 查询/设置一次性操作 | 服务 | 同步请求-应答 |
| 长时间运行的任务（导航、抓取） | 动作 | 支持反馈和取消 |
| 节点配置参数 | 参数 | 运行时可调，支持YAML文件 |

# 八、TF2坐标变换

TF2是ROS2中的坐标变换库，用于管理多个坐标系之间的关系。

## 发布TF变换

```python
from geometry_msgs.msg import TransformStamped
import tf2_ros

class TfBroadcaster(Node):
    def __init__(self):
        super().__init__('tf_broadcaster')
        self.br = tf2_ros.TransformBroadcaster(self)
        self.timer = self.create_timer(0.1, self.broadcast_tf)

    def broadcast_tf(self):
        t = TransformStamped()
        t.header.stamp = self.get_clock().now().to_msg()
        t.header.frame_id = 'world'
        t.child_frame_id = 'robot_base'
        t.transform.translation.x = 1.0
        t.transform.rotation.w = 1.0
        self.br.sendTransform(t)
```

## 监听TF变换

```python
import tf2_ros

class TfListener(Node):
    def __init__(self):
        super().__init__('tf_listener')
        self.tf_buffer = tf2_ros.Buffer()
        self.tf_listener = tf2_ros.TransformListener(self.tf_buffer, self)
        self.timer = self.create_timer(1.0, self.lookup_tf)

    def lookup_tf(self):
        try:
            trans = self.tf_buffer.lookup_transform('world', 'robot_base', rclpy.time.Time())
            self.get_logger().info(f'Translation: {trans.transform.translation}')
        except tf2_ros.TransformException as e:
            self.get_logger().warn(f'变换失败: {e}')
```
